// 代码示例

if(data.result.length) {
    this.userList = data.result;
} else {
  notice('提示', '暂无数据', 'warning');
}
// 所以 在开始点击查询使用者信息的接口时候 先清空一次userList 在查接口
this.userList.length = 0;

if (!value || !property) {

if (document.defaultView && document.defaultView.getComputedStyle) {
